create table "avalanche" (
    "id" bigint primary key,
    "createTime" timestamp not null,
    "updateTime" timestamp not null,
    "extId" varchar(8) not null,
    "viewable" boolean not null,
    "submitterExp" integer not null,
    "lat" double precision not null,
    "lng" double precision not null,
    "areaName" varchar(128) not null,
    "avyDate" date not null,
    "sky" integer not null,
    "precip" integer not null,
    "elevation" integer not null,
    "aspect" integer not null,
    "angle" integer not null,    
    "avyType" integer not null,
    "avyTrigger" integer not null,
    "avyInterface" integer not null,
    "rSize" double precision not null,
    "dSize" double precision not null,
    "caught" integer not null,
    "partiallyBuried" integer not null,
    "fullyBuried" integer not null,
    "injured" integer not null,
    "killed" integer not null,
    "modeOfTravel" integer not null,
    "comments" text not null,
    "kmlCoords" text not null,
    "submitterId" bigint not null
  );
create sequence "s_avalanche_id";
-- indexes on avalanche
create unique index "idx2f5d05cf" on "avalanche" ("extId");
  
create table "avalanche_image" (
    "id" bigint primary key,
    "createTime" timestamp not null,
    "avyExtId" varchar(8) not null,
    "filename" varchar(128) not null,
    "mimeType" varchar(20) not null,
    "size" integer not null,
    "bytes" bytea not null
);
create sequence "s_avalanche_image_id";
-- indexes on avalanche_image
create index "idx76e30961" on "avalanche_image" ("avyExtId");

create table "app_user" (
    "id" bigint primary key not null,
    "createTime" timestamp not null,
    "email" varchar(128) not null
);
create sequence "s_app_user_id";
-- indexes on app_user
create unique index "idx2a600595" on "app_user" ("email");

create table "app_user_role" (
    "id" bigint primary key not null,
    "createTime" timestamp not null,
    "name" varchar(128) not null
);
create sequence "s_app_user_role_id";

create table "app_user_role_assignment" (
    "userId" bigint not null,
    "roleId" bigint not null
);

-- foreign key constraints :
alter table "avalanche" add constraint "avalancheFK1" foreign key ("submitterId") references "app_user"("id");
alter table "app_user_role_assignment" add constraint "app_user_role_assignmentFK2" foreign key ("userId") references "app_user"("id");
alter table "app_user_role_assignment" add constraint "app_user_role_assignmentFK3" foreign key ("roleId") references "app_user_role"("id");

-- composite key indexes :
alter table "app_user_role_assignment" add primary key ("userId","roleId");

-- column group indexes :
create unique index "idxdb8b0ccf" on "avalanche_image" ("avyExtId","filename");
